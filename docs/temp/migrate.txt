enum DivisionLevel {
  country
  state
  city
  district
  ward
}

model AdministrativeDivision {
  id        Int     @id @default(autoincrement())
  code      String? @unique // mã hành chính (nếu có)
  name      String
  level     DivisionLevel

  parentId  Int?
  parent    AdministrativeDivision? @relation("DivisionHierarchy", fields: [parentId], references: [id])
  children  AdministrativeDivision[] @relation("DivisionHierarchy")

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([level])
  @@index([level, id])
  @@index([parentId])
}
